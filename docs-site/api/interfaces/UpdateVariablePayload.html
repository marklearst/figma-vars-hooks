    Interface UpdateVariablePayload

    Payload for updating properties of an existing Figma variable.

    Remarks

    Use with the useUpdateVariable hook to specify fields to update. All fields are optional, allowing partial updates.

    Example

    import type { UpdateVariablePayload } from '@figma-vars/hooks';

    const updatePayload: UpdateVariablePayload = {
      name: 'Updated Color Name',
      description: 'Updated description',
    }
    interface UpdateVariablePayload {
        codeSyntax?: Record<string, string>;
        description?: string;
        hiddenFromPublishing?: boolean;
        name?: string;
        scopes?: VariableScope[];
    }
